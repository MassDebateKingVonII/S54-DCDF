Arithmetic Operators
Operator Description Example
+ Addition 
                                                
                                                    let x = 5;
                                                    let y = 2;
                                                    console.log(x + y); // Output is 7
- Subtraction 
                                                
                                                    let x = 5;
                                                    let y = 2;
                                                    console.log(x - y); // Output is 2
* Multiplication 
                                                
                                                    let x = 5;
                                                    let y = 3;
                                                    console.log(x * y); // Output is 5 x 3 = 15
** Exponentiation 
                                                
                                                    let x = 5;
                                                    let y = 3;
                                                    console.log(5 ** 3); // 5^3 = 125
/ Division 
                                                
                                                    let x = 10;
                                                    let y = 2;
                                                    console.log(x/y); // Output is 10 รท 2 = 5
% Modulus (Remainder) 
                                                
                                                    let x = 10;
                                                    let y = 3;
                                                    console.log(x%y); // Output is 10 รท 3 = 3R1, hence 1
++ Increment 
                                                
                                                let x = 5;
                                                x++;
                                                console.log(x);  // 6
-- Decrement 
                                                
                                                    let x = 5;
                                                    x--;
                                                    console.log(x); // Output is 4
Source: W3Schools
